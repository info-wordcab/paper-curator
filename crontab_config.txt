# ArXiv Paper Curator Cron Jobs
# Add these lines to your crontab (run 'crontab -e' to edit)

# Main paper collection and digest - runs at 9 AM on Mon, Wed, Fri
0 9 * * 1,3,5 /usr/bin/python3 /path/to/arxiv_curator.py >> /path/to/logs/cron.log 2>&1

# Reply checking - runs every 6 hours
0 */6 * * * /usr/bin/python3 /path/to/arxiv_curator.py --check-replies-only >> /path/to/logs/cron_replies.log 2>&1

# Weekly backup of database - Sundays at 2 AM
0 2 * * 0 cp /path/to/papers_database.db /path/to/backups/papers_database_$(date +\%Y\%m\%d).db

# Monthly cleanup of old PDFs - First day of month at 3 AM
0 3 1 * * find /path/to/pdfs -mtime +90 -type f -delete
